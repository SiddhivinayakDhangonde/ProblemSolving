using System;
using System.Collections.Generic;
class HelloWorld {
  public static void Main() {
      int count=0;
      int max=0;
    string name="abcabcbb";
    int n=name.Length;
    List<string> li=new List<string>();
    List<string> uq=new List<string>();
    for(int i=0;i<n;i++){
        string a="";
        for(int j=i;j<n;j++){
              a=a+name[j];
            li.Add(a);
        }
    }
    foreach(string item in li){
      if(CheckDuplicate(item)){
          uq.Add(item);
      }
    }
     foreach(string item in uq){
            int s= item.Length;
            if(s>max){
                count++;
                max=s;
            }
      }
      Console.Write("Max uniqueck substring count with max length"+max);
  }
      static  bool CheckDuplicate(string s){
        HashSet<char> hs=new HashSet<char>();
        foreach(char item in s){
            if(!hs.Add(item)){
                return false;
            }
        }
        return true;
    
  }
     
}
