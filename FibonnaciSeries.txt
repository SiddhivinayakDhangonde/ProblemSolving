using System;
class HelloWorld {
    static int Fabonnacie(int num){
          if(num==0){
              return 0;
          }
          else if(num==1){
              return 1;
          }
          return Fabonnacie(num-1)+Fabonnacie(num-2);
      } 
  static void Main() {
    Console.WriteLine("Enter the Number");
    int num=Convert.ToInt32(Console.ReadLine());
    for(int i=0;i<num;i++){
        Console.Write(Fabonnacie(i)+" ");
    }
  }
}
