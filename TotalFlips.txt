
using System;
class HelloWorld {
    public static int Flips(string word){
        int count=0;
        for(int i=0;i<word.Length-1;i+=2){
            if(word[i]!=word[i+1]){
                count++;
            }
        }
        return count;
    }
  static void Main() {
   
      string word=Console.ReadLine();
      int a=Flips(word);
      Console.WriteLine("total flips:-"+a);
  }
}
